
        START   101
        READ    N
        MOVER   BREG ='1'
        MOVEM   BREG TERM
AGAIN   MULT    BREG TERM
        MOVER   CREG ='2'
        LTORG
        ADD     CREG ='1'
        MOVEM   CREG TERM
        COMP    CREG N
        BC      LE AGAIN
        DIV     BREG ='2'
        MOVEM   BREG RESULT
        PRINT   RESULT
        STOP
N       DS      1
RESULT  DS      1
ONE     DC      '1'
TERM    DS      1
TWO     DC      '2'
        END
